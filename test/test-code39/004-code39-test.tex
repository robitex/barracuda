Test 4: uint costructor

\directlua{
barracuda = require "barracuda"
local c39, err = barracuda:get_barcode_class()
                          :new_encoder("code39")
assert(not err, err)
local err
local symbol
symbol, err = c39:from_uint(0, {text_hpos="center"})
assert(not err, err)

local canvas = barracuda:new_canvas()
symbol:append_ga(canvas)

local drv, err = barracuda:load_driver("native")
assert(not err, err)
drv:ga_to_hbox(canvas, "mybox")
}
\box\mybox
